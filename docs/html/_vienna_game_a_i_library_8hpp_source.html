703 std::vector<Vec2ui> findPreprocessedPath(Vec2ui start, Vec2ui target)
704 {
705 NodeData& startNode = getNode(start);
706 NodeData& targetNode = getNode(target);
707
708 ui32 startRegionID = startNode.regionID;
709 ui32 targetRegionID = targetNode.regionID;
710
711 // If start and target nodes are in the same region, call A* to find the shortest path
712 if (startRegionID == targetRegionID)
713 {
714 return findPath(start, target);
715 }
716
717 // Check if there is at least one precomputed path from the start node stored in `m_adjList`
718 if (m_adjList[getIndex(startNode.pos)].size() == 0)
719 {
720 std::cout << "Cannot find path!" << std::endl;
721 return {};
722 }
723
724 ui32 startNodeIndex = getIndex(startNode.pos);
725 std::vector<Vec2ui> pathToRegion = m_adjList[startNodeIndex][targetRegionID];
726
727 // Check if there is a precomputed path from the start node to the region in which the target node is
728 if (pathToRegion.size() == 0)
729 {
730 std::cout << "No path found to region!" << std::endl;
731 return {};
732 }
733
734 // Retrieve the last node from the newly found path and check whether it is the target node
735 Vec2ui pathEndNode = pathToRegion[pathToRegion.size() - 1];
736 if (pathEndNode == target)
737 {
738 // Reached destination
739 return pathToRegion;
740 }
741
742 // If the end node of the path is not the target, get the shortest path between it and the target
743 std::vector<Vec2ui> pathWithinRegion = findPath(pathEndNode, target);
744 if (pathWithinRegion.size() == 0)
745 {
746 std::cout << "No path found inside region!" << std::endl;
747 return {};
748 }
749
750 // Combine the two paths together: (start -> endNode) + (endNode -> target)
751 pathToRegion.insert(pathToRegion.end(), pathWithinRegion.begin(), pathWithinRegion.end());
752
753 return pathToRegion;
754 }

1032 std::vector<Vec2ui> AStar(Vec2ui start, Vec2ui target, std::vector<NodeData>& nodes)
1033 {
1034 std::vector<i32> parents(nodes.size(), -1);
1035
1036 for (NodeData& node : nodes)
1037 {
1038 node.g = INFINITY;
1039 node.h = INFINITY;
1040 }
1041
1042 std::priority_queue<NodeData, std::vector<NodeData>, NodeDataComparator> openSet;
1043
1044 ui32 startNodeIndex = getIndex(start);
1045 ui32 targetNodeIndex = getIndex(target);
1046
1047 nodes[startNodeIndex].g = 0.0f;
1048
1049 openSet.push(nodes[startNodeIndex]);
1050
1051 while (openSet.size() > 0)
1052 {
1053 ui32 currentIndex = getIndex(openSet.top().pos);
1054
1055 while (openSet.size() > 0 && currentIndex == getIndex(openSet.top().pos))
1056 {
1057 openSet.pop();
1058 }
1059
1060 if (currentIndex == targetNodeIndex)
1061 {
1062 std::vector<Vec2ui> shortestPath;
1063
1064 while (currentIndex != -1)
1065 {
1066 shortestPath.push_back(nodes[currentIndex].pos);
1067 currentIndex = parents[currentIndex];
1068 }
1069
1070 std::reverse(shortestPath.begin(), shortestPath.end());
1071 return shortestPath;
1072 }
1073
1074 std::vector<ui32> neighbors = m_neighbors[currentIndex];
1075 for (ui32 neighborIndex : neighbors)
1076 {
1077 NodeData& neighbor = nodes[neighborIndex];
1078 if (neighbor.state == NodeState::OBSTRUCTABLE)
1079 continue;
1080
1081 f32 tentativeG = nodes[currentIndex].g + euclidean(neighbor.pos, nodes[currentIndex].pos);
1082
1083 if (tentativeG < neighbor.g)
1084 {
1085 parents[neighborIndex] = getIndex(nodes[currentIndex].pos);
1086 neighbor.g = tentativeG;
1087 neighbor.h = euclidean(neighbor.pos, nodes[targetNodeIndex].pos);
1088
1089 openSet.push(nodes[neighborIndex]);
1090 }
1091 }
1092 }
1093
1094 return {};
1095 }
